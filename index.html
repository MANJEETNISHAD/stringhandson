Question-2:Count the Heads

var Count_Occ = (s) => 
        {
          let str = s.split('').sort().join('');
            let ans = "";
            for(let i=0;i<(str.length);i++){
              let count = 0;
              for(let j=0;j<(str.length);j++){
                if(str[i]===str[j]&&i>j){
                  break;
                }
                if(str[i]===str[j]){
                  count++;
                }
              }
              if(count>1){
                ans+=str[i]+count;
              }
            }
            return ans;  
        };

Question-6:Find the Winner

 
var Game_Winner = (S) => 
{
  let A=0;
  let D=0;
  for(let i=0;i<(S.length);i++){
    if("A"==S[i]){
      A++
    }
    else{
      D++
    }
  }
  if(A==D){
    return "Draw"
  }
  else if(A>D){
    return "Aditya"
  }
  else{
    return "Danish"
  }
}
